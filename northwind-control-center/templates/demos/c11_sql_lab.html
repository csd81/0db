Actual #temp tables are session-scoped and drop when the connection closes — they cannot be demonstrated via separate HTTP requests. The equivalent derived-table (subquery in FROM) solution produces identical results.
Temp table pattern (reference only — not runnable) 
-- Create temp table from a query
SELECT e.employeeid, lastname,
    year(orderdate) as ev,
    month(orderdate) as mo,
    count(orderid) as no_of_orders
INTO #tt
FROM employees e LEFT OUTER JOIN orders o
    ON e.employeeid=o.employeeid
GROUP BY e.employeeid, lastname,
    year(orderdate), month(orderdate)

-- Then query the temp table
SELECT lastname,
    STR(avg(cast(no_of_orders as float)),15,2)
    as avg_monthly
FROM #tt
GROUP BY employeeid, lastname
ORDER BY avg_monthly DESC

DROP TABLE #tt  -- cleanup
Derived table equivalent (runnable ↓) 
-- Inline subquery replaces #tt
SELECT src.lastname,
    STR(AVG(CAST(src.no_of_orders AS FLOAT)),
        15, 2) as avg_monthly_orders
FROM (
    SELECT e.employeeid, e.lastname,
        YEAR(o.orderdate) as ev,
        MONTH(o.orderdate) as mo,
        COUNT(o.orderid) as no_of_orders
    FROM employees e
    LEFT OUTER JOIN orders o
        ON e.employeeid=o.employeeid
    GROUP BY e.employeeid, e.lastname,
        YEAR(o.orderdate), MONTH(o.orderdate)
) AS src  -- alias is REQUIRED
GROUP BY src.employeeid, src.lastname
ORDER BY avg_monthly_orders DESC
